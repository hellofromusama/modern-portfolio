RESEARCH-VERIFIED CSS (add to globals.css):
  /* Opt into cross-document (MPA) view transitions - progressive enhancement.
     Unsupported browsers simply navigate normally, no animation. */
  @view-transition {
    navigation: auto;
  }

  /* MUST respect reduced motion - kill all VT animation when requested */
  @media (prefers-reduced-motion: reduce) {
    ::view-transition-group(*),
    ::view-transition-old(*),
    ::view-transition-new(*) {
      animation-duration: 0s !important;
      animation-delay: 0s !important;
    }
  }
